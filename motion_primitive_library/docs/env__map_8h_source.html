90  decimal_t traverse_primitive(const Primitive<Dim> &pr) const {
91  decimal_t max_v = 0;

93  if (pr.max_vel(i) > max_v) max_v = pr.max_vel(i);

113  if (!potential_map_.empty()) {

115  c += dt * (potential_weight_ * potential_map_[idx] +
